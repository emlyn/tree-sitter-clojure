=====================================
Syntax quote
=====================================

`(foo ~x)
`(foo ~@y)
`(foo z#)

---

(program
  (syntax_quote
    (list
      (symbol)
      (unquote (symbol))))

  (syntax_quote
    (list
      (symbol)
      (unquote_splice (symbol))))

  (syntax_quote
    (list
      (symbol)
      (gensym)))
  )
